// --- ЛОГІН ТА ІНІЦІАЛІЗАЦІЯ --- window.login = async function() { const user = document.getElementById('username').value.trim(); const pass = document.getElementById('password').value.trim(); if(!user || !pass) { alert("Введи дані!"); return; } myName = user; isYustya = (user === "Yustya" && pass === "YustyaLove"); const userRef = ref(db, "users/" + user); const snapshot = await get(userRef); // Беремо картинку з канвасу як "запасний варіант" const c = document.getElementById('avatarCanvas'); const fallbackAvatar = c.toDataURL('image/png'); if(isYustya) { myMoney = 999999; avatarBase64 = "yustya"; // Спеціальний маркер для рушія } else if (snapshot.exists()) { if (snapshot.val().password !== pass) { alert("Невірний пароль!"); return; } myMoney = snapshot.val().money || 0; // ФІКС ПОМИЛКИ: Якщо це старий акаунт без збереженої картинки, // беремо поточну з екрану, щоб не було undefined! avatarBase64 = snapshot.val().avatar || fallbackAvatar; } else { avatarBase64 = fallbackAvatar; if(avatarBase64.length < 5000) { alert("Завантаж фотку, ти ж не Юстя!"); return; } myMoney = 200; set(userRef, { password: pass, money: myMoney, avatar: avatarBase64 }); } document.getElementById('login-screen').style.display = 'none'; document.getElementById('game-container').style.display = 'block'; resizeCanvas(); window.addEventListener('resize', resizeCanvas); // Підключаємося до онлайну const myPlayerRef = ref(db, "players/" + myName); // Тепер skin гарантовано не буде undefined set(myPlayerRef, { x: player.x, y: player.y, skin: avatarBase64 }); onDisconnect(myPlayerRef).remove(); // Видаляємо персонажа при виході updateUI(); startGameLoop(); setupOnlinePlayers(); };
